\subsection*{基于RDMA的对象传输性能测试}

\begin{frame}
	\frametitle{优化实现的传输性能}

	\begin{block}{总结}
		\begin{itemize}
			\item 32KB时切换协议：内存注册 vs. 内存拷贝
			\item 小对象：RDMA Send > RDMA Read $\approx $ Socket，1KB~15\%提升
			\item 大对象：RDMA Read > RDMA Send > Socket，4MB~7倍吞吐
		\end{itemize}
	\end{block}
